8    import uz.umarxon.serializableexample.Utils.Constants
9    import uz.umarxon.serializableexample.model.User
10   
11   class EditActivity : AppCompatActivity() {
12   
13       private var tv: TextView = findViewById(R.id.myTv)
14   
15       @SuppressLint("SetTextI18n")
16       override fun onCreate(savedInstanceState: Bundle?) {
17           super.onCreate(savedInstanceState)
18           setContentView(R.layout.activity_edit)
19   
20           /**
21            * Bu yerda companion object bo'lgan intent orqali
22            * jo'natgan user class ma'lumotimizni olamiz
23            *
24            * Bu yerda kelgan user malumotlimizni tekshirib olamiz
25            * agar u yaratish uchun yoki update qilish uchun bo'lsa create yoki update degan btn larni
26            * visible qilamiz agar u shunchaki ko'rish (read) uchun bo'lsa btn larni visiblity.GONE
27            * qilib qo'yamiz
28            *
29            * Agar  [findNavController]  ishlatilganida
30            * var myUser:User = arguments.getSerializableExtra("keyUser") as User
31            * ko'rinishida bo'lar edi.
32            * */
33   
34           val myUser: User = intent.getSerializableExtra("keyUser") as User
35   
36           val type = intent.getIntExtra("intent_type",0)
37   
38           when(type){
39   
40               Constants.TYPE_CREATE -> {
41                   //agar malumot yaratilishi kerak bo'lsa shunga tayyorgarlik ko'riladi
42                   tv.text = "${myUser.name} \n ${myUser.number}"
43               }
44   
45               Constants.TYPE_UPDATE -> {
46                   //agar malumot yangilanishi kerak bo'lsa shunga tayyorgarlik ko'riladi
47                   tv.text = "${myUser.name} \n ${myUser.number}"
48               }
49   
50               Constants.TYPE_READ -> {
51                   //agar malumot shunchaki (read) o'qilishi kerak bo'lsa shunga tayyorgarlik ko'riladi
52                   tv.text = "${myUser.name} \n ${myUser.number}"
53               }
54   
55           }
56   
57       }
58   }
